`[slug].js`:

```js
/**
 * ページコンポーネントで使用する値を用意する
 */
export async function getStaticProps({ params }) {
  const content = await readContentFile({ fs, slug: params.slug });

  return {
    props: {
      ...content,
    },
  };
}
```
